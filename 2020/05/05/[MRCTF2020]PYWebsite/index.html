function str2binl(str)
{
  var bin = Array();
  var mask = (1 << 8) - 1;//mask=255
  for(var i = 0; i < str.length * 8; i += 8)
    bin[i>>5] |= (str.charCodeAt(i / chrsz) & mask) << (i%32);
  return bin;
}
